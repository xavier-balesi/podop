    Scheduler()

    Fake delay scheduler, CPU greedy.

    The game scheduler is not based on real timestamps or real delays, enabling the scheduling of the next game turn directly after the previous one. Indeed, the primary objective of the game is the construction of a deep reinforcement learning model, and the more matches we run, the better.

    def scheduleabs(
    self,
    timestamp: int,
    action: Callable[..., Any],
    *args: Any,
    **kwargs: Any,
) -> None:
    """Schedule a new event.

    Events scheduled for the same time will be executed in a random order.
    This information might be useful for the player's strategy.
    Executing the event means executing `action(*args, **kwargs)`.

    Params:
        timestamp: Absolute timestamp in milliseconds
        action: Function to be executed
        *args: Sequence holding the positional arguments for action
        **kwargs: Dictionary holding the keyword arguments for action
    """
    # If "the random order" (cf.docstring) is a problem, just add a sequence number in the _Task
    # just after the _Task.start_ts member and use itertools.count() to generate it.
    simplified_action = partial(action, *args, **kwargs)
    heappush(self._tasks, _Task(start_ts=timestamp, action=simplified_action))
    log.debug(
        f"scheduling at {ts_to_str(timestamp)} action {action} with args={args} and kwargs={kwargs}",
    )

    def set_timestamp(self, timestamp: int) -> None:
    """Accelerate (or reverse) the time with an absolute timestamp."""
    while self._tasks and (action_ts := self._tasks[0].start_ts) <= timestamp:
        task: _Task = heappop(self._tasks)
        func = task.action.func
        self.ts = action_ts
        log.debug(
            f"executing the action {func.__name__ if hasattr(func, "__name__") else func} planned for {ts_to_str(action_ts)}",
        )
        task.action()
    self.ts = timestamp

    def abort_tasks(self) -> None:
    """Abort all scheduled tasks."""
    if not self._tasks:
        return
    log.debug(f"aborting {len(self._tasks)} tasks")
    self._tasks[:] = []

    @contextmanager
def from_timestamp_contextmanager(self, ts: int):
    """Accelerate (or reverse) the time temporarily with a contextmanager.

    Allow changing the current timestamp so that child schedules do not depend on future time shifting.
    """
    orig_ts = self.ts
    # change now timestamp
    self.ts = ts
    try:
        yield
    finally:
        # restore now timestamp
        self.ts = orig_ts

    def from_timestamp_decorator(self, ts: int):
    """Accelerate (or reverse) the time temporarily with a decorator.

    Allow changing the csecondsurrent timestamp so that child schedules do not depend on future time shifting.
    """

    def _from_timestamp_decorator(f):
        @wraps(f)
        def _from_timestamp_wrapper(*args, **kwargs):
            with self.from_timestamp_contextmanager(ts):
                return f(*args, **kwargs)

        return _from_timestamp_wrapper

    return _from_timestamp_decorator

    def ts_to_str(  # NB: method and docstring generated by ChatGPT.
    timestamp: int,  # in ms
) -> str:
    """Convert a timestamp in milliseconds to a formatted string representing hours, minutes, seconds, and milliseconds.

    Arguments:
        timestamp (int): The timestamp in milliseconds.

    Returns:
        A formatted string representing the timestamp in the format \
        "HH:MM:SS.sss", where HH represents hours, MM represents minutes, \
        SS represents seconds, and sss represents milliseconds.

    Examples:
        >>> ts_to_str(1 * 3600 * 1000 + 2 * 60 * 1000 + 3 * 1000 + 4)
        '01:02:03.004'
        >>> ts_to_str(2 * 60 * 1000 + 3 * 1000 + 4)
        '02:03.004'
        >>> ts_to_str(3 * 1000 + 4)
        '03.004'
        >>> ts_to_str(4)
        '00.004'
    """  # NB: examples tested with: pdm run python -m doctest src/back/scheduler.py
    milliseconds = int(timestamp % 1000)
    seconds = int((timestamp / 1000) % 60)
    minutes = int((timestamp / (1000 * 60)) % 60)
    hours = int((timestamp / (1000 * 60 * 60)) % 24)

    # Format hours, minutes, seconds, and milliseconds
    formatted_time: str = ""
    if hours > 0:
        formatted_time += f"{hours:02d}:"
    if minutes > 0 or hours > 0:
        formatted_time += f"{minutes:02d}:"
    formatted_time += f"{seconds:02d}"

    # Add milliseconds if any.
    if milliseconds > 0:
        formatted_time += f".{milliseconds:03d}"

    return formatted_time
