    Counts ¤

    Bases: RootModel[Counter[str]]

    Count all resources of the game at a given timestamp.

    Info

    Counts are transmitted to the frontend periodically according to the framerate parameter.

    This class is a mixed between:

    1. collections.Counter to stores counts easily
    2. addict.Dict (or Munch, Easydict, AttrDict, Prodict) 3rd party libraries to access dict values as attributes
    3. pydantic models to serialize the object to JSON
    Example

    Counts can be obtained with two syntaxes.

    def update(self, transaction: Transaction) -> None:
    """Update the timestamp and the counts of game resources.

    Args:
        transaction (Transaction): The new transaction to take into account.
    """
    root = self.root
    root["ts"] = transaction.ts
    for add in transaction.add:
        root[add.type] += add.value if type(add) is Money else 1
    for remove in transaction.remove:
        root[remove.type] -= remove.value if type(remove) is Money else 1

    def get_ressource(
    self,
    ressource_name: Literal["foo", "bar", "foobar"],
    lock: bool = True,
) -> IncrIdRessources | None:
    """Get a ressource with the possibility to `lock` the ressource.

    Args:
        ressource_name (Literal["foo", "bar", "foobar"]): The name of the ressource to get.
        lock (bool, optional): Lock the ressource to prevent other robots from using it. Defaults to True.
    """
    ressources = getattr(self, ressource_name + "s")
    for ressource in ressources:
        if ressource.lock:
            continue
        if lock:
            ressource.lock = True
        return ressource
    return None

    def get_ressources(
    self,
    ressource_name: Literal["foo", "bar", "foobar"],
    count: int,
    lock=True,
) -> list[IncrIdRessources]:
    """Get `count` ressources with the possibility to `lock` the ressources.

    Args:
        ressource_name (Literal["foo", "bar", "foobar"]): The name of the ressource to get.
        count (int): The number of ressources to get.
        lock (bool, optional): Lock the ressource to prevent other robots from using it. Defaults to True.
    """
    returned_ressources = []
    ressources = getattr(self, ressource_name + "s")
    for ressource in ressources:
        if ressource.lock:
            continue
        if lock:
            ressource.lock = True
        returned_ressources.append(ressource)
        if len(returned_ressources) == count:
            break

    # Revert the lock when the count is not there.
    if len(returned_ressources) != count:
        for ressource in returned_ressources:
            ressource.lock = False
        return []

    return returned_ressources

    def on_new_transaction(self, transaction: Transaction) -> None:
    """Update the inventory after each new [`Transaction`][back.models.transaction.Transaction]."""
    for model in transaction.add:
        match model:
            case Foo():
                self.foos.append(model)
            case Bar():
                self.bars.append(model)
            case FooBar():
                self.foobars.append(model)
            case Money():
                self.money += model
            # Uncovered branch coverage:
            # https://github.com/pytest-dev/pytest-cov/issues/533
            case Robot():  # pragma: no cover
                self.robots.append(model)

    for model in transaction.remove:
        match model:
            case Foo():
                self.foos.remove(model)
            case Bar():
                self.bars.remove(model)
            case FooBar():
                self.foobars.remove(model)
            case Money():  # pragma: no cover
                self.money -= model

    def __init_subclass__(cls, **kwargs):
    """Verify at import time that all subclasses have mandatory fields.

    For example, verify the presence of the discriminator `type` necessary for parsing json representation of resources.

    !!! danger "What happens without this verification at import time?"

        If the game finally mint/forge a new resource, we will have this error at runtime:
         > Unable to extract tag using discriminator 'type'

    """
    # __init_subclass__ avoid having to use a metaclass in this use case.
    super().__init_subclass__(**kwargs)
    if "type" not in cls.__dict__:
        raise TypeError(f"The class {cls.__name__} must have a 'type' field.")
