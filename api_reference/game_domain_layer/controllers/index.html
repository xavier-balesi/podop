    def _on_new_transaction(self, transaction: Transaction) -> None:
    """Receive a transaction made by robots.

    For example, a robot trades euros and foos to buy a new robot.
    Transactions reflects new game states.

    The game engine historicizes the transactions and propose to the player
    to pilot its robots. The player will have to elaborate a good strategy
    to accumulate new robots as fast as possible.
    """
    super()._on_new_transaction(transaction)

    # Observe new robots.
    for model in transaction.add:
        match model:
            case Robot():
                model.subscribe(Transaction, self._on_new_transaction)

    # Control when the inventory gets updated.
    self._inventory.on_new_transaction(transaction)

    # Eventually stop the game.
    if len(self._inventory.robots) >= game_config.max_robots:
        self.running = False
        return

    # Once the inventory is updated, propose to the player to play, make orders.
    self._player.trade_in_live()

    async def run(self) -> None:
    """Start the game engine. The game ends when reaching [max_robots][back.config.GameConfig.max_robots]."""
    # Player plays for the first time.
    self._player.trade_in_live()

    while self.running:
        self._start_next_turn()
        # Even if turn_interval=0, we need to wait to not block the main asyncio loop and
        # to abort the program on Ctrl+C.
        await asyncio.sleep(game_config.turn_interval / 1000)

    # Abort planned tasks when the game is over.
    scheduler.abort_tasks()

    def _start_next_turn(self) -> None:
    """Perform a turn by executing all scheduled tasks.

    Generated sub-tasks will be executed in the next turn.
    """
    if busy_until := scheduler.busy_until:
        # Execute all tasks by jumping to the last task timestamp.
        scheduler.set_timestamp(busy_until)
    else:  # pragma: no cover (dead code is no bug...)
        log.critical("⛔ The game has deadlock... Aborting.")
        log.critical(f"Final counts: {self.get_counts()}")
        self.running = False

    def get_counts_history(self) -> CountsHistory:
    """Return a list of resource counts over time."""
    counts_history: CountsHistory = CountsHistory()
    for transaction in self._transactions:
        if len(counts_history) == 0:
            new_counts = Counts(ts=0, foo=0, bar=0, foobar=0, robot=0, money=0)  # type: ignore
            new_counts.update(transaction)
            counts_history.append(new_counts)
        else:
            last_counts = counts_history[-1]
            new_counts = last_counts.model_copy(deep=True)
            new_counts.update(transaction)
            assert new_counts.ts == transaction.ts
            if last_counts.ts == new_counts.ts:
                counts_history[-1] = new_counts
            else:
                counts_history.append(new_counts)

    # We can display the returned data in color with:
    # from rich import print_json
    #
    # print_json(data=counts_history.dict(exclude_none=True))
    return counts_history

    def trade_in_live(self) -> None:
    """Play each time the game changes."""
    if (game := self._game) is None:
        log.warning("The game is detached fom the player.")
        return
    inventory = game.get_inventory()
    for robot in inventory.robots:
        if robot.action != Action.WAITING_FOR_ORDER:
            continue

        # It's always possible to mine foo or bar.
        possible_actions = [robot.mine_foo, robot.mine_bar]

        # Can we forge a foobar?
        if inventory.get_foo(lock=False) and inventory.get_bar(lock=False):  # type: ignore
            possible_actions.append(robot.forge_foobar)

        # Can we sell a random count of foobar?
        sell_count = randint(1, game_config.sell_foobar_max_count)
        if inventory.get_foobars(count=sell_count, lock=False):  # type: ignore
            possible_actions.append(partial(robot.sell_foobar, count=sell_count))

        # Can we buy a robot?
        if inventory.money >= game_config.money_for_robot and inventory.get_foos(  # type: ignore
            count=game_config.foo_for_robot,
            lock=False,
        ):
            possible_actions.append(robot.buy_robot)

        # Choose a random action that respect game rules.
        random_action = choice(possible_actions)
        log.debug(
            f"player choose {random_action} while {inventory}",
            extra={"🤖": robot.id},
        )
        random_action()

    Robot(inventory, *args, **kwargs)

    Bases: Provider, IncrIdRessource

    Hard-working robot at the mercy of the player.

    Attributes:

    Name Type Description
    action Action

    The current action/state of the robot.
    previous_action Action

    The previous action/state of the robot.
    Implementation strategy

    The game engine is based solely on these two attributes.

    We could have used several variables:

    • the location of the robot (on the road, foo mine)
    • a state (available or busy)
    • an action (move, mine, forge or sell)

    at the risk of having inconsistencies between the variables.

    As the previous action is updated automatically when the current action is changed, almost the entire game engine depends on a single variable, thus limiting risk.

    def _forge_foobar_callback(self, foo, bar) -> None:
    """Assembling the foobar is ended. Is it a success? If not, the foobar and the foo are trashed."""
    random = randint(1, 100)
    if random <= game_config.forge_foobar_success_rate:
        self._notify_transaction(
            add=[FooBar.build()],
            remove=[foo, bar],
        )
    else:
        bar.lock = False
        self._notify_transaction(remove=[foo])

    @action_wrapper
def buy_robot(self) -> None:
    """Buy a robot instantly if there is enough money and foo(s)."""
    self.action = Action.SELL_FOOBAR
    inventory = self._inventory
    if inventory.money >= game_config.money_for_robot and (  # type: ignore
        foos := inventory.get_foos(count=game_config.foo_for_robot)  # type: ignore
    ):
        # Thankfully the transactions using money is not delay, so we don't have to lock the money.
        self._notify_transaction(
            add=[Robot.build(inventory=inventory)],
            remove=[Money(value=game_config.money_for_robot), *foos],
        )
    else:
        if inventory.money < game_config.money_for_robot:
            self._log.debug(
                f"not enough money to buy a robot: {inventory.money.value}/{game_config.money_for_robot} while {inventory}",
            )
        else:
            self._log.debug(
                f"not enough foo(s) to buy a robot ({game_config.money_for_robot} money + {game_config.foo_for_robot} foo are needed) while {inventory}",
            )
            for foo in foos:
                foo.lock = False
        self.action = Action.WAITING_FOR_ORDER
        return

    def action_wrapper(f):
    """Decorate all robot's actions with this decorator.

    This decorator refuses the order from the [player][back.controllers.player.RandomStrategyPlayer]
    or accept it with a delay in case the robot have to move to another activity.
    """

    @wraps(f)
    def _action_wrapper(self, *args, **kwargs):
        if self.action != Action.WAITING_FOR_ORDER:
            raise RobotBusyError(
                f"Robot {self} can't {f.__name__} because it is not waiting for an order.",
            )

        # Robot is at the good place, no need to move.
        if self.previous_action.name.lower() == f.__name__:
            return f(self, *args, **kwargs)

        # The robot have to move before executing the action.
        self.action = Action.MOVING
        after_moving_ts = scheduler.ts + game_config.move_duration
        scheduler.schedule(
            game_config.move_duration,
            # Future "child" schedules have to be executed directly after the move
            # and not depend on future time shifting.
            scheduler.from_timestamp_decorator(after_moving_ts)(f),
            self,
            *args,
            **kwargs,
        )
        return None

    return _action_wrapper
