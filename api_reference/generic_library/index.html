    def dict_merge(dct: dict[Any, Any], merge_dct: dict[Any, Any]) -> None:
    """Recursive `dict` merge.

    Inspired by `dict.update()`, instead of
    updating only top-level keys, dict_merge recurse down into dicts nested
    to an arbitrary depth, updating keys. The ``merge_dct`` is merged into
    ``dct``.

    https://gist.githubusercontent.com/angstwad/bf22d1822c38a92ec0a9/raw/dbb0d3e7b0640be0237bd72058ddd8f84788b583/dict_merge.py

    :param dct: :class:`dict` onto which the merge is executed
    :param merge_dct: dct merged into dct
    :return: None
    """
    for k, v in merge_dct.items():
        if k in dct:
            original_value = dct[k]
            if isinstance(original_value, dict) and isinstance(v, dict):
                # Recursive call to dict merge in order to merge sub-dict found in parent dict.
                dict_merge(original_value, v)
                continue
        dct[k] = v

def format_time_ns(s: int, ns: int) -> str:
    """Return the current time as GMT iso8601 formatted text with nano seconds."""
    ct = gmtime(s)
    txt = strftime("%Y-%m-%dT%H:%M:%S", ct)

    @staticmethod
def format_tb(tb):
    """Format python traceback for json logging."""
    stack_summary: StackSummary = StackSummary.extract(
        walk_tb(tb),
        capture_locals=True,
    )
    frame: FrameSummary  # noqa: F842
    # Ignore frame.locals to respect security requirements.
    return [
        {
            "file": frame.filename,
            "instruction": frame.line,
            "line": frame.lineno,
            "method": frame.name,
        }
        for frame in stack_summary
    ]

    @staticmethod
def format_exception(e: type[BaseException]):
    """Format python exception for json logging."""
    formatted_exception = {
        "name": e.__class__.__name__,
        "message": repr(e),
        # We could pass in parameter record.exc_info[2] to avoid reading a private attribute
        # but, after __context__ and __cause__ usage, one more or one less...
        "extendedStackTrace": JsonFormatter.format_tb(e.__traceback__),
    }

    # It would have been preferable to reuse TracebackException.format() method to not use
    # private members __cause__ and __context__ but instead of using it on the exception varable,
    # we would use it on the TracebackException instance to override TracebackException.stack.format()
    # for all causes.
    if e.__cause__ is not None:
        cause = e.__cause__
    elif e.__context__ is not None and not e.__suppress_context__:
        cause = e.__context__
    else:
        cause = None
    if cause is not None:
        formatted_exception["cause"] = JsonFormatter.format_exception(cause)

    return formatted_exception

    def format(self, record):
    """Build the json that will be logged.

    Properties are set in same order than text logs.
    """
    # don't use systematically an OrderedDict for orjson compatibility
    log_record = {} if self.use_orjson else OrderedDict()

    # Field names are specified in https://ecomptes.jira.com/wiki/spaces/DEV/pages/2228060165/Logging+-+Fields+list.
    # Order matters to respect "debug" LOG_FORMAT.
    log_record["level"] = _level_name_to_json[record.levelname]
    log_record["levelAsInteger"] = record.levelno
    log_record["thread"] = record.threadName
    created_ns = time_ns()
    s, ns = divmod(created_ns, 10**9)
    # The field "time" is supposed to be used by Fluentd to compute @timestamp field,
    # but it seems nano seconds aren't supported.
    log_record["time"] = self.format_time_ns(s, ns)
    # Fallback on "instant" fields to compute @timestamp field.
    log_record["instant"] = {"epochSecond": s, "nanoOfSecond": ns}

    extra = join_extra(record)
    extra = translate_extra(extra)
    log_record.update(extra)
    log_file = (
        log_record.pop("real_pathname")
        if log_record.get("real_pathname", None)
        else record.pathname
    )
    log_line = (
        log_record.pop("real_lineno")
        if log_record.get("real_lineno", None)
        else record.lineno
    )

    log_record["loggerName"] = record.name
    log_record["logFile"] = log_file
    log_record["logLine"] = log_line
    log_record["message"] = record.getMessage()

    exc_info = record.exc_info
    if exc_info:
        log_record["thrown"] = self.format_exception(exc_info[1])

    # Python2.7 doesn't have stack_info=True argument when logging.
    # `record.stack_info` is a str and it's impossible to format it like the exception tracebacks.
    if sys.version_info.major == 3 and record.stack_info:
        log_record["logStackTrace"] = self.formatStack(record.stack_info)

    return self.json_serializer(log_record)

    def log_level_adapter(level: str) -> int:
    """Convert LOG_LEVEL=2, or debug, or DEBUG to the standard log level 20."""
    if level.isdigit():
        level_int = int(level) * 10
    else:
        level_int = logging.getLevelName(level.upper())
        if not isinstance(level_int, int):
            raise ValueError(
                f"{level!r} is not a valid log level, use {tuple(LOG_COLORS.keys())}",
            )
    return level_int

    def set_pkg_name(pkg_name) -> None:
    """Set the prefix that will be used in environment variables like prefix_LOG_LEVEL."""
    global DEBUG, LOG_BACKEND, LOG_FILE, LOG_FILTER, LOG_FORMAT, LOG_LEVEL, LOG_COLOR  # pylint: disable=global-variable-undefined

    def join_extra(record):
    """Merge all extra together.

    The ones set while configuring the logs,
    and the ones set with :func:`update_log_extra`,
    and the ones set while logging.
    """
    extra = deepcopy(global_extra)
    # Need to deepcopy _global_extra_context to not update sub-dictionaries by reference on second dict_merge
    extra_from_context = deepcopy(_global_extra_context.get())
    dict_merge(extra, extra_from_context)
    if isinstance(record, UnmergedExtraLogRecord):
        dict_merge(extra, record.extra)
    elif DEBUG:
        # when logging object record.msg is not yet converted to a string
        if record.name == "root":
            record.msg = (
                str(record.msg)
                + " (extra are missing as using root logger instead of logging.getLogger())"
            )
        else:
            record.msg = (
                str(record.msg)
                + " (extra are missing as configure_logs() is called after the first logging.getLogger())"
            )
    return extra

    def setup_handlers() -> None:  # pylint: disable=too-many-branches
    """Define the log format, the log backend, ..."""
    root_logger = logging.getLogger()
    root_logger.setLevel(LOG_LEVEL)

    formatter = JsonFormatter() if LOG_FORMAT == "json" else ExtraTextFormatter()

    if "std" in LOG_BACKEND:  # accept stdout or stderr
        file = getattr(sys, LOG_BACKEND)
        if LOG_FORMAT == "rich":
            # don't install rich by default, let the developer install it manually if desired
            from rich.console import (  # pylint: disable=import-error,import-outside-toplevel
                Console,
            )
            from rich.logging import (  # pylint: disable=import-error,import-outside-toplevel
                RichHandler,
            )

            # As throw_low_level_logs() had redirected stdout, we have to precise its file descriptor
            # to avoid "OSError: [Errno 25] Inappropriate ioctl for device".
            console = Console(file=file, width=os.get_terminal_size(0).columns)
            handler = RichHandler(
                console=console,
                rich_tracebacks=True,
                tracebacks_show_locals=True,
            )
        else:
            handler = logging.StreamHandler(file)
            if LOG_FORMAT != "json" and (
                LOG_COLOR == "always"
                or (
                    LOG_COLOR == "auto"
                    and sys.platform != "win32"
                    and handler.stream.isatty()
                )
            ):
                formatter = ColorTextFormatter()
            handler.setFormatter(formatter)
    elif LOG_BACKEND == "file":
        if not LOG_FILE:
            print("please set the LOG_FILE environment variable")  # noqa: T201
            return
        handler = logging.FileHandler(LOG_FILE)
        handler.setFormatter(formatter)
    elif LOG_BACKEND == "syslog":
        handler = SysLogHandler(address="/dev/log")
        handler.setFormatter(formatter)
    elif LOG_BACKEND == "detect":
        for handler in root_logger.handlers:
            handler.setFormatter(formatter)
        return
    else:
        print("LOG_BACKEND should be stdout, stderr, file or syslog")  # noqa: T201
        return

    handler.is_mine = True
    root_logger.addHandler(handler)

    # replace the default exception handler to replace sys.stderr with the newly configured LOG_BACKEND
    sys.excepthook = except_hook

        # apply the log filter only if there is more precision than previous log filter
        # Ex: 1:package < 3:package.module
        if len(local_filter_str) >= max_path_length:
            level = local_filter_level
            max_path_length = len(local_filter_str)
    # don't change the log level of logger created before configure_logs() call
    if level is not None:
        orig_set_level(self, level)

    def setup_log_filter() -> None:
    """Override the log level of some modules according to prefix_LOG_FILTER."""
    if not LOG_FILTER:
        return

    # initialize log_filter_list
    try:
        for elt in LOG_FILTER.split("|") if LOG_FILTER else []:
            (filter_level, filter_str) = elt.split(":")
            log_filter_list.append(
                (filter_str, re.compile(filter_str), log_level_adapter(filter_level)),
            )
    except ValueError:
        print('try this format: prefix_LOG_FILTER="2:.*module1.*|5:.*module2.*"')  # noqa: T201
        raise

    # support prefix_LOG_FILTER on new loggers
    patch("logging.getLogger", new=new_get_logger).start()

    # force usage of prefix_LOG_FILTER even if someone intends to change it
    patch.object(logging.Logger, "setLevel", new=new_set_level).start()

    # support prefix_LOG_FILTER on old loggers
    for name in logging.root.manager.loggerDict:  # pylint: disable=no-member
        new_set_level(logging.getLogger(name))

    def unconfigure_logs() -> None:
    """Revert the log configuration."""
    if logging.getLogger is not orig_get_logger:
        logging.getLogger = orig_get_logger
        logging.Logger.setLevel = orig_set_level

    update_log_extra()

    # remove handlers previously added

    def throw_low_level_logs() -> None:
    """Throw out low level logs made by some modules implemented in C/C++ that use printf().

    Redirect all language C/C++ logs from stdout (file descriptor number 1) to /dev/null
    but keep python logs on stdout.

    Caution:
        - it breaks breakpoint() history, so use prefix_DEBUG=1 environment variable when using a python shell
        - incompatible with pytest

    This is a workaround made for Pulsar, the time the log configuration is added to pulsar-client package:
    https://github.com/apache/pulsar/issues/4234.
    We prefer this workaround instead of forking the pulsar-client package.

    In the future, if we fork pulsar-client to keep warnings and errors, we will start to enable log4cxx
    by adding `cmake . -DUSE_LOG4CXX` in `pulsar-client-cpp/docker/build-wheel-file-within-docker.sh`.

    """
    if is_fd_protected():
        return

    fd_no_1 = sys.stdout.fileno()  # pythonic way to do "fd_no_1 = 1"

    # Set another fd in sys.stdout that is linked to the initial destination of stdout:
    # - step1: create a new fd that points to the initial destination of stdout
    #          (that can be found with `readlink /proc/self/fd/1`)
    stdout_copy = os.dup(fd_no_1)
    # - step2: by saving a io.TextIOWrapper in sys.stdout,
    #          python prints will now go to `readlink /proc/self/fd/1` by using the new fd
    sys.stdout = os.fdopen(stdout_copy, "w")

    # Now the file descriptor N°1 is no more used by python,
    # redirect it to /dev/null to lose language C logs done with printf().
    with open("/dev/null", "w", encoding="utf-8") as stdout_redirect:
        os.dup2(stdout_redirect.fileno(), fd_no_1)

    @classmethod
def load_from_files(cls, *files_path) -> Any:
    """Instantiate the class with the content of multiple yaml files.

    :param files_path: files to merge and load (respecting order)
    """
    full_config: dict[str, Any] = {}
    for file_path in files_path:
        dict_merge(full_config, cls._load_yaml_content(file_path))

    try:
        return cls(**full_config)
    except ValidationError as e:
        raise YamlLoaderError(
            f"Wrong config file format [{files_path}]: {e}",
        ) from e

    @staticmethod
def _load_yaml_content(file_path: str) -> dict[str, Any]:
    """Open and parse file.

    :param file_path: yaml file path
    """
    try:
        with open(file_path, encoding="utf-8") as f:
            return yaml.safe_load(f)
    except FileNotFoundError as e:
        raise YamlLoaderError(f"Cannot find config file {file_path} !") from e
    except (ParserError, ScannerError) as e:
        raise YamlLoaderError(f"Cannot parse config file {file_path} !") from e
