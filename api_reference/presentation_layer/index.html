    @classmethod
@lru_cache
def load(cls):
    """Load the configuration files designated by `CONFIG_PATH` and `SECRET_CONFIG_PATH` environment variables and cache the configuration in memory.

    ???+ note

        If no environment variables are set, loads the default configuration.
    """
    config_paths: list[str] = []
    for key in ("CONFIG_PATH", "SECRET_CONFIG_PATH"):
        if key in os.environ:
            config_paths.append(os.environ[key])
    # Try to instantiate the model directly in case all fields have default values.
    config = cls.load_from_files(*config_paths) if config_paths else cls()
    log.debug(f"loaded configuration: {config!r}")
    return config

    def cancel_all_tasks(exclude_current=False) -> None:
    """Cancel all asyncio tasks to force exiting on Ctrl+C.

    Necessary only because of the game infinite loop.
    TODO: find cleaner solution.
    """
    all_tasks = asyncio.all_tasks()
    if exclude_current and (current_task := asyncio.current_task()):  # pragma: no cover
        all_tasks.remove(current_task)
    for task in all_tasks:
        task.cancel()

    async def main() -> None:
    """Call this entry point from the Dockerfile to launch the backend."""
    app_config: ApplicationConfig = ApplicationConfig.load()

    for app, port in (
        ("back.api.routes:app", api_port := app_config.api.port),
        (
            "back.api.monitoring:app",
            monitoring_port := app_config.api.monitoring_port,
        ),
    ):
        # WARNING: when providing the default log_config, it disabled all existing loggers, and we loose some logs.
        config = Config(app, host="0.0.0.0", port=port, log_config=None)
        servers.append(MultiServer(config=config, on_shutdown=exit_application))

    log.info(
        f"the backend is ready and listen on {api_port} (API) and {monitoring_port} (monitoring)",
    )
    # By using a dedicated port for monitoring features (healthcheck, metrics, changing log level, etc.),
    # and not exposing it, the frontend can't have access to these features.
    # Relying on an API Gateway to block access to the `/monitoring` path poses higher risks.
    await asyncio.gather(*(server.async_run() for server in servers))

    def configure_back_logs() -> None:  # pragma: no cover (only using pytest log handlers)
    """Configure the backend logs."""
    # TODO: test features of open source loggers to avoid maintaining an home made logger.
    #  Is json format is supported? Readable format like provided by RichHandler
    #    (https://rich.readthedocs.io/en/latest/logging.html?highlight=handler#logging-handler)?
    #  Is is easy to set a log context (globally, for a thread or coroutines)?
    #  Can we filter logs for debugging? Etc.
    #  Elastic Common Schema (ECS) logger propose to standardize field names and provide the json format.
    #    https://www.elastic.co/guide/en/ecs-logging/python/master/installation.html
    #  structlog propose a lot of features and is a mature project:
    #    https://www.structlog.org/en/stable/contextvars.html
    #  picologging from Microsoft is the last in date and it's performance oriented but features seems limited:
    #    https://microsoft.github.io/picologging/index.html
    configure_logs(
        json_translations=OrderedDict(
            {
                # Emoji are easy to read while debugging, but avoid them in production logs.
                # NB: can be tested with LOG_LEVEL=debug LOG_FORMAT=json pdm back.
                "🤖": "robot",
            },
        ),
    )
    # Limit builtin and 3rd party modules logs.
    for module in ("asyncio", "urllib3", "uvicorn"):
        logging.getLogger(module).setLevel(logging.WARNING)

    def catch_disconnections(func):
    """Catch client disconnections with a decorator.

    It's to avoid having an enormous try/except or multiple ones for each interaction with the WebSocket.
    """

    @functools.wraps(func)
    async def _catch_disconnections_wrapper(*args, **kwargs):
        try:
            return await func(*args, **kwargs)
        except WebSocketDisconnect as e:
            log.warning(f"Client disconnected: {e!r}", exc_info=True)
            return None

    return _catch_disconnections_wrapper

    @app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket) -> None:
    """Start a new game by connecting to a websocket.

    We can also debug the backend without frontend by doing:

    ```bash
    python -m websockets 'ws://localhost:8080/ws'
    ```

    The game runs as fast as possible. Eventually slow it down by using the `turn_interval` config.
    Spying the game by sampling the game resources at a given framerate (see `framerate` config).
    Game information are communicated in JSON format by a websocket.
    A complete resource history is sent when the game is over.
    """
    log.info("Client connected")
    await websocket.accept()
    game = Game()

    @catch_disconnections
    async def notify_front() -> None:
        while game.running:
            await asyncio.sleep(1.0 / api_config.framerate)

            # Sampling number of game resources on each frame.
            front_msg: str = game.get_counts().json()

            # Using a websocket to eventually pilot the trading strategy manually from the web client.
            await websocket.send_text(front_msg)

        # Send the complete resource history when the game is over.
        counts_history: CountsHistory = game.get_counts_history()
        # TODO: timeit orjson.dumps(model.dict(exclude_none=True))
        front_msg = counts_history.json(exclude_none=True)
        await websocket.send_text(front_msg)

    # Run the game and spy the game in parallel, by sampling the inventory.
    await asyncio.gather(game.run(), notify_front())

    game.destroy()
